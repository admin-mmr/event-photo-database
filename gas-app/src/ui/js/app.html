* just before the app.html include. * * We navigate window.top (not window) because the GAS web app renders inside * an iframe on googleusercontent.com; only the outer script.google.com frame * properly routes query-parameter changes back through doGet. */ function navigate(action) { var base = (window.APP_URL || window.location.href).split('?')[0]; window.top.location.href = base + '?action=' + action; } // Intercept all href="?action=..." links so they go through navigate() // instead of trying to load a googleusercontent.com URL directly. document.addEventListener('DOMContentLoaded', function () { document.querySelectorAll('a[href]').forEach(function (link) { var href = link.getAttribute('href') || ''; if (!href.includes('action=')) return; link.addEventListener('click', function (e) { e.preventDefault(); var qs = href.startsWith('?') ? href.slice(1) : (href.split('?')[1] || ''); var action = new URLSearchParams(qs).get('action'); if (action) navigate(action); }); }); }); // ── Confirmation helper ──────────────────────────────────────────────────── /** * Replaces a button with an inline confirmation prompt. * Returns a Promise that resolves true (confirmed) or false (cancelled). * * @param {Element} triggerBtn The button that was clicked * @param {string} promptText e.g. "Deactivate Jane Doe?" */ function confirmInline(triggerBtn, promptText) { return new Promise(function (resolve) { var parent = triggerBtn.parentNode; var saved = triggerBtn.outerHTML; var prompt = document.createElement('span'); prompt.className = 'confirm-prompt'; prompt.innerHTML = promptText + ' ' + ' ' + ''; parent.replaceChild(prompt, triggerBtn); prompt.querySelector('button:first-of-type').addEventListener('click', function () { parent.innerHTML = saved; // Re-upgrade MDL on restored elements if (window.componentHandler) componentHandler.upgradeDom(); resolve(true); }); prompt.querySelector('button:last-of-type').addEventListener('click', function () { parent.innerHTML = saved; if (window.componentHandler) componentHandler.upgradeDom(); resolve(false); }); }); }
